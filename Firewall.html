Добавление приложений в фаервол брандмауера через коммандную строку

netsh advfirewall firewall add rule - комманда Добавить правило фаервола в брандмауер с именем
name="10000 tekla_dstv2dxf.exe" - имя правила
action=allow - действие=разрешить
action=block - действие=блокировать
dir=IN - исходящее соединение
dir=out - входящее соединение
program="полный путь к программе" - путь к программе
enable=yes - включить правило
enable=no - не включать правило
